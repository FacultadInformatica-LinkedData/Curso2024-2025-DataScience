## **7-1. Consultas referentes a Lighting**

1. Listar todos los elementos de alumbrado y su tipo.
2. Elementos de alumbrado promovidos por un promotor específico.
3. Listar los motivos de alumbrado utilizados en cada distrito.
4. Elementos de alumbrado registrados como nuevos.
5. Obtener información completa de un alumbrado.
6. Listar todos los barrios y los elementos de alumbrado que pertenecen a ellos.
7. Alumbrados con longitud mayor que 500.
8. Promedio de longitud de los alumbrados en cada distrito.
9. Consultar todos los alumbrados con un motivo específico.

from rdflib import Graph

# Definición de prefijos
prefixes = """
PREFIX tm: <http://turismo.mad.es/cultural/resource/>
PREFIX xsd: <http://www.w3.org/2001/XMLSchema#>
"""

# Consultas SPARQL
queries_lighting = [
    # 1. Listar todos los elementos de alumbrado y su tipo
    prefixes + """
    SELECT ?lighting ?type
    WHERE {
        ?lighting a tm:Lighting .
        ?lighting tm:hasType ?type .
    }
    LIMIT 10
    """,
    # 2. Elementos de alumbrado promovidos por un promotor específico
    prefixes + """
    SELECT ?lighting
    WHERE {
        ?lighting a tm:Lighting .
        ?lighting tm:hasPromoter "AG OBRAS Y EQUIPAMIENTOS" .
    }

    """,
    # 3. Listar los motivos de alumbrado utilizados en cada distrito
    prefixes + """
    SELECT ?district ?motive
    WHERE {
      ?lighting a tm:Lighting .
      ?lighting tm:hasDistrict ?district .
      ?lighting tm:hasMotive ?motive .
    }
    LIMIT 10
    """,
    # 4. Elementos de alumbrado registrados como nuevos
    prefixes + """
    SELECT ?lighting
    WHERE {
        ?lighting a tm:Lighting .
        ?lighting tm:isNew "SI" .
    }
    """,
    # 5. Obtener información completa de un alumbrado
    prefixes + """
    SELECT ?property ?value
    WHERE {
        ?lighting a tm:Lighting .
        ?lighting ?property ?value .
    }
    LIMIT 10
    """,
    # 6. Listar todos los barrios y los elementos de alumbrado que pertenecen a ellos
    prefixes + """
    SELECT ?neighborhood ?lighting
    WHERE {
        ?lighting a tm:Lighting .
        ?lighting tm:hasNeighborhood ?neighborhood .
    }
    LIMIT 10
    """,
    # 7. Alumbrados con longitud mayor que 500
    prefixes + """
    SELECT ?lighting ?length
    WHERE {
        ?lighting a tm:Lighting .
        ?lighting tm:hasLength ?length .
        FILTER(xsd:integer(?length) > 500)
    }
    LIMIT 10
    """,
    # 8. Promedio de longitud de los alumbrados en cada distrito
    prefixes + """
    SELECT ?district (AVG(xsd:integer(?length)) AS ?avgLength)
    WHERE {
        ?lighting a tm:Lighting .
        ?lighting tm:hasDistrict ?district .
        ?lighting tm:hasLength ?length .
    }
    GROUP BY ?district
    LIMIT 10

    """,
    # 9. Consultar todos los alumbrados con un motivo específico
    prefixes + """
    SELECT ?lighting ?district ?neighborhood ?type
    WHERE {
        ?lighting a tm:Lighting .
        ?lighting tm:hasMotive ?motive .
        ?lighting tm:hasDistrict ?district .
        ?lighting tm:hasNeighborhood ?neighborhood .
        ?lighting tm:hasType ?type .
        FILTER(?motive = "COLGANTES")
    }
    LIMIT 10
    """
]

# Consultas SPARQL (idénticas a las del archivo RDF)
for i, query in enumerate(queries_lighting, 1):
    print(f"Consulta {i}:")

Consultar salas que están en un barrio específico
Contar el número de salas en cada distrito
Encontrar salas accesibles que tienen transporte cercano
Obtener información completa de una dirección
Agrupar barrios por distrito y contar cuántos hay en cada uno
Salas que están en una calle específica
Salas dentro de un radio de una ubicación (latitud, longitud)

from rdflib import Graph
# Consultas referentes a monument
# Definición de prefijos
prefixes = """
PREFIX tm: <http://turismo.mad.es/cultural/resource/>
PREFIX xsd: <http://www.w3.org/2001/XMLSchema#>
"""

# Consultas SPARQL
queries_venue = [
    # 1. Listar todas las salas y sus direcciones
    prefixes + """
    SELECT ?venue ?address
    WHERE {
        ?venue a tm:Venue .
        ?venue tm:hasAddress ?address .
    }
    """,
    # 2. Consultar salas que están en un barrio específico
    prefixes + """
    SELECT ?venue
    WHERE {
        ?venue a tm:Venue .
        ?venue tm:Venue ?venue .
        ?address tm:hasNeighborhood tm:chopera .
    }
    """,
    # 3. Contar el número de salas en cada distrito
    prefixes + """
    SELECT ?district (COUNT(?venue) AS ?numvenues)
    WHERE {
        ?venue a tm:Venue .
        ?venue tm:hasAddress ?address .
        ?address tm:hasDistrict ?district .
    }
    GROUP BY ?district
    """,
    # 4. Encontrar salas accesibles que tienen transporte cercano
    prefixes + """
    SELECT ?venue
    WHERE {
        ?venue a tm:Venue .
        ?venue tm:hasAccessibility ?accessibility .
        ?venue tm:hasNearbyTransport ?transport .
        FILTER(?accessibility = "1")
    }
    """,
    # 5. Obtener información completa de una dirección
    prefixes + """
    SELECT ?property ?value
    WHERE {
        tm:4408284471551 ?property ?value .
    }
    """,
    # 6. Agrupar barrios por distrito y contar cuántos hay en cada uno
    prefixes + """
    SELECT ?district (COUNT(?venue) AS ?numvenue)
    WHERE {
        ?venue a tm:Venue .
        ?venue tm:hasAddress ?address .
        ?address tm:hasDistrict ?district .
    }
    GROUP BY ?district
    """,
    # 7. Salas que están en una calle específica
    prefixes + """
    SELECT ?venue
    WHERE {
        ?venue a tm:Venue .
        ?venue tm:hasAddress ?address .
        ?address tm:hasStreet tm:principe .
    }
    """,
    # 8. Salas dentro de un radio de una ubicación (latitud, longitud)
    prefixes + """
    SELECT ?venue
    WHERE {
        ?venue a tm:Venue .
        ?venue tm:hasAddress ?address .
        ?address tm:hasLatitude ?lat .
        ?address tm:hasLongitude ?long .
        FILTER (
            (xsd:decimal(?lat) - 40.3776)*(xsd:decimal(?lat) - 40.3776) +
            (xsd:decimal(?long) - (-3.6229))*(xsd:decimal(?long) - (-3.6229)) <= 0.01
        )
    }
    """,
    # 9. Salas que están en un barrio específico
    prefixes + """
    SELECT ?venue
    WHERE {
        ?venue a tm:Venue .
        ?venue tm:hasAddress ?address .
        ?address tm:hasNeighborhood tm:recoletos .
    }
    """
]
# Consultas SPARQL

7-6. Consultas conjuntas
Museos y luces en el mismo barrio.
Museos con luces cercanas
Contar recursos por tipo en cada barrio.
Museos con información de iluminación relacionada
Barrios con más museos y luces.

# Definición de prefijos comunes
prefixes = """
PREFIX tm: <http://turismo.mad.es/cultural/resource/>
PREFIX xsd: <http://www.w3.org/2001/XMLSchema#>
"""

# Consultas SPARQL
queries_conjuntas = [
    # 1. Monumentos o museos en el mismo barrio
    prefixes + """
    SELECT DISTINCT ?barrio ?museo ?monumento
    WHERE {
        ?museo a tm:Museum .
        ?museo tm:hasAddress ?direccion .
        ?direccion tm:hasNeighborhood ?barrio .

        ?monumento a tm:Monument .
        ?monumento tm:hasAddress ?monumentodireccion .
        ?monumentodireccion tm:hasNeighborhood ?barrio .

    }
    LIMIT 15
    """,
    # 2. Museos y luces en la misma calle
    prefixes + """
    SELECT DISTINCT ?neighborhood ?museo ?luces
    WHERE {
      ?museo a tm:Museum .
      ?museo tm:hasAddress ?direccion .
      ?direccion tm:hasNeighborhood ?neighborhood .

      ?luces a tm:Lighting .
      ?luces tm:hasNeighborhood ?neighborhood .
    }
    LIMIT 15
    """,


    # 3. Contar recursos por tipo en cada barrio
    prefixes + """
    SELECT ?neighborhood (COUNT(?museum) AS ?numMuseums) (COUNT(?lighting) AS ?numLightings)
    WHERE {
        OPTIONAL {
            ?museum a tm:Museum .
            ?museum tm:hasAddress ?museumAddress .
            ?museumAddress tm:hasNeighborhood ?neighborhood .
        }
        OPTIONAL {
            ?lighting a tm:Lighting .
            ?lighting tm:hasAddress ?lightingAddress .
            ?lightingAddress tm:hasNeighborhood ?neighborhood .
        }
    }
    GROUP BY ?neighborhood
    """,

    # 4. Museos con información de iluminación relacionada
    prefixes + """
    SELECT ?museum ?museumName ?lighting ?lightingName
    WHERE {
        ?museum a tm:Museum .
        ?museum tm:hasName ?museumName .
        ?museum tm:hasAddress ?museumAddress .
        ?museumAddress tm:hasNeighborhood ?neighborhood .

        ?lighting a tm:Lighting .
        ?lighting tm:hasName ?lightingName .
        ?lighting tm:hasAddress ?lightingAddress .
        ?lightingAddress tm:hasNeighborhood ?neighborhood .
    }
    """,

    # 5. Barrios con más museos y luces
    prefixes + """
    SELECT ?neighborhood (COUNT(?museum) + COUNT(?lighting) AS ?totalResources)
    WHERE {
        OPTIONAL {
            ?museum a tm:Museum .
            ?museum tm:hasAddress ?museumAddress .
            ?museumAddress tm:hasNeighborhood ?neighborhood .
        }
        OPTIONAL {
            ?lighting a tm:Lighting .
            ?lighting tm:hasAddress ?lightingAddress .
            ?lightingAddress tm:hasNeighborhood ?neighborhood .
        }
    }
    GROUP BY ?neighborhood
    ORDER BY DESC(?totalResources)
    """
]

# Consultas SPARQL (idénticas a las del archivo RDF)
for i, query in enumerate(queries_conjuntas, 1):
    #if i == 2:

      for row in combined_graph.query(query):
